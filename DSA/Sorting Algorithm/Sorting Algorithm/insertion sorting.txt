#include <stdio.h>

int main() {
    int a[6] ={15,16,6,8,5,3};
    int n = sizeof(a)/sizeof(a[0]);
    int i ,j,temp;
    for(i = 1;i<n;i++)
    {
        temp =a[i];
        j=i-1;
        while(j>=0 && a[j]>temp)
        {
            a[j+1]=a[j];
            j--;
        }
        a[j+1]=temp;
    }
    for(int k=0;k<n;k++)
    {
        printf("%d ",a[k]);
    }

    return 0;
}
